1. 定义国际化（i18n）架构设计与实现规范
2. 明确多语言支持的范围与管理策略
3. 规范多时区场景下的时间处理规则
4. 定义多区域部署架构与数据主权策略
5. 建立 GDPR、数据跨境、隐私保护等合规框架
6. 确保产品在全球范围内的合法合规运营

## 四、国际化架构

### 4.1 i18n 设计原则

#### 4.1.1 核心设计原则

| 原则 | 说明 | 实现要求 |
|------|------|----------|
| 文本外置原则 | 所有用户可见文本必须外置到语言包 | 禁止硬编码文本字符串 |
| 格式分离原则 | 数字、日期、货币等格式与文本分离 | 使用格式化库而非手动拼接 |
| 文化中立原则 | 代码逻辑不假设特定文化习惯 | 使用 ICU 标准处理复数/性别 |
| 渐进增强原则 | 先实现基础功能，再增强本地化 | 保持核心功能可运行 |
| 右到左支持原则 | 支持阿拉伯语等右到左语言布局 | 使用 CSS 逻辑属性 |

```typescript
  'en-US': {
    date: {
      year: 'numeric',

    },
    number: {
      decimal: '.',
      grouping: ',',
    },
    currency: {
      currency: 'USD',
      style: 'currency',
    },
  },
  'ja-JP': {
    date: {
      year: 'numeric',
      month: 'long',
      day: 'numeric',
    },
    number: {
      decimal: '.',
      grouping: ',',
    },
    currency: {
      currency: 'JPY',
      style: 'currency',
    },
  },
};

class FormatService {
  private intl: IntlShape;
  
  constructor(locale: string) {
    this.intl = createIntl({
      locale,
      ...locales[locale] || locales['zh-CN'],
    });
  }
  
  formatDate(date: Date | number, format: 'short' | 'medium' | 'long' = 'medium'): string {
    return this.intl.formatDate(date, {
      year: 'numeric',
      month: format === 'short' ? 'numeric' : format === 'medium' ? 'short' : 'long',
      day: 'numeric',
    });
  }
  
  formatNumber(value: number, options?: Intl.NumberFormatOptions): string {
    return this.intl.formatNumber(value, options);
  }
  
  formatCurrency(value: number, currency?: string): string {
    return this.intl.formatNumber(value, {
      style: 'currency',
      currency: currency || this.intl.locale.split('-')[1],
    });
  }
  
  formatPercent(value: number): string {
    return this.intl.formatNumber(value, { style: 'percent' });
  }
  
  formatList(values: string[], style: 'long' | 'short' | 'narrow' = 'long'): string {
    return this.intl.formatList(values, { style });
  }
}
```

#### 5.1.3 语言包版本管理

```typescript
interface LanguageBundle {
  locale: string;
  version: string;
  module: string;
  translations: Record<string, any>;
  status: 'draft' | 'translation' | 'review' | 'published';
  updatedAt: Date;
  updatedBy: string;
}

interface TranslationKey {
  key: string;
  module: string;
  description?: string;
  maxLength?: number;
  placeholders?: string[];
  status: 'active' | 'deprecated' | 'removed';
}

class LanguageBundleService {
  async getBundles(locale: string, module?: string): Promise<LanguageBundle[]> {
    const query = this.db.createQueryBuilder('language_bundles')
      .where('locale = :locale', { locale })
      .andWhere('status = :status', { status: 'published' });
    
    if (module) {
      query.andWhere('module = :module', { module });
    }
    
    return query.orderBy('version', 'DESC').getMany();
  }
  
  async updateBundle(bundle: LanguageBundle): Promise<void> {
    const newVersion = this.incrementVersion(bundle.version);
    
    await this.db.transaction(async (manager) => {
      await manager.save('language_bundles', {
        ...bundle,
        version: newVersion,
        updatedAt: new Date(),
      });
      
      await this.invalidateCache(bundle.locale, bundle.module);
      
      await this.publishEvent('bundle_updated', {
        locale: bundle.locale,
        module: bundle.module,
        version: newVersion,
      });
    });
  }
  
  async checkMissingKeys(): Promise<TranslationKey[]> {
    const allKeys = await this.getAllTranslationKeys();
    const missing: TranslationKey[] = [];
    
    for (const locale of SUPPORTED_LOCALES) {
      const existingKeys = await this.getExistingKeys(locale);
      
      for (const key of allKeys) {
        if (!existingKeys.includes(key.key)) {
          missing.push({
            ...key,
            locale,
          });
        }
      }
    }
    
    return missing;
  }
}
```

#### 5.2.2 语言切换实现

```typescript
class LanguageSwitchService {
  async switchLanguage(userId: bigint, newLocale: string): Promise<void> {
    if (!this.isSupportedLocale(newLocale)) {
      throw new Error(`不支持的语言: ${newLocale}`);
    }
    
    const user = await this.userService.findById(userId);
    
    await this.db.transaction(async (manager) => {
      await manager.update('users', {
        id: userId,
      }, {
        locale: newLocale,
        updatedAt: new Date(),
      });
      
      await manager.save('user_preferences', {
        userId,
        key: 'language',
        value: newLocale,
        updatedAt: new Date(),
      });
    });
    
    await this.invalidateUserCache(userId);
    
    await this.publishEvent('language_changed', {
      userId,
      oldLocale: user.locale,
      newLocale,
    });
  }
  
  getLanguageFromRequest(request: Request): string {
    const sources = [
      request.query['lang'] as string,
      request.cookies['i18next'],
      request.headers['accept-language']?.split(',')[0],
      this.config.defaultLocale,
    ];
    
    for (const source of sources) {
      if (source && this.isSupportedLocale(source)) {
        return source;
      }
    }
    
    return this.config.defaultLocale;
  }
  
  isSupportedLocale(locale: string): boolean {
    return SUPPORTED_LOCALES.includes(locale);
  }
}
```

```typescript
class TimeDisplayService {
  constructor(
    private timezoneService: TimezoneService,
    private userId?: bigint
  ) {}
  
  formatListTime(utcDate: string | Date | number): string {
    const userTimezone = this.getUserTimezone();
    const localDate = this.timezoneService.fromUTC(utcDate, userTimezone);
    return localDate.format('YYYY/MM/DD HH:mm');
  }
  
  formatDetailTime(utcDate: string | Date | number): string {
    const userTimezone = this.getUserTimezone();
    const localDate = this.timezoneService.fromUTC(utcDate, userTimezone);
    return localDate.format('YYYY年MM月DD日 HH:mm');
  }
  
  formatRelative(utcDate: string | Date | number): string {
    const now = dayjs.utc();
    const target = dayjs.utc(utcDate);
    const diffSeconds = target.diff(now, 'second');
    
    if (Math.abs(diffSeconds) < 60) {
      return diffSeconds >= 0 ? '刚刚' : '刚刚';
    } else if (Math.abs(diffSeconds) < 3600) {
      const minutes = Math.floor(Math.abs(diffSeconds) / 60);
      return diffSeconds >= 0 ? `${minutes}分钟后` : `${minutes}分钟前`;
    } else if (Math.abs(diffSeconds) < 86400) {
      const hours = Math.floor(Math.abs(diffSeconds) / 3600);
      return diffSeconds >= 0 ? `${hours}小时后` : `${hours}小时前`;
    } else {
      const days = Math.floor(Math.abs(diffSeconds) / 86400);
      return diffSeconds >= 0 ? `${days}天后` : `${days}天前`;
    }
  }
  
  formatDateRange(
    startUtc: string | Date | number,
    endUtc: string | Date | number
  ): string {
    const userTimezone = this.getUserTimezone();
    const start = this.timezoneService.fromUTC(startUtc, userTimezone);
    const end = this.timezoneService.fromUTC(endUtc, userTimezone);
    
    if (start.isSame(end, 'day')) {
      return `${start.format('YYYY/MM/DD HH:mm')} - ${end.format('HH:mm')}`;
    }
    
    return `${start.format('YYYY/MM/DD HH:mm')} - ${end.format('YYYY/MM/DD HH:mm')}`;
  }
```

```typescript
      return userPref;

    await this.invalidateUserCache(userId);
  }

      throw new Error(`无效的时区: ${timezone}`);
    }
    
    await this.saveTenantConfig(tenantId, 'timezone', timezone);
    await this.invalidateTenantCache(tenantId);
  }

    const tzHeader = request.headers['x-user-timezone'];
    if (tzHeader && this.timezoneService.isValidTimezone(tzHeader)) {
```

```typescript
interface PrivacyProtectionConfig {
  piiFields: string[];
  sensitiveFields: string[];
  maskRules: {
    [field: string]: {
      type: 'partial' | 'full' | 'none';
      prefixLength?: number;
      suffixLength?: number;
      maskChar?: string;
    };
  };
  encryptionRequired: boolean;
  accessLogRequired: boolean;
}

const DEFAULT_PRIVACY_CONFIG: PrivacyProtectionConfig = {
  piiFields: [
    'name', 'phone', 'email', 'id_card', 
    'bank_account', 'address', 'company_name'
  ],
  sensitiveFields: [
    'password', 'id_card', 'bank_account', 'security_code'
  ],
  maskRules: {
    phone: { type: 'partial', prefixLength: 3, suffixLength: 4 },
    email: { type: 'partial', prefixLength: 3, suffixLength: 0 },
    id_card: { type: 'partial', prefixLength: 3, suffixLength: 4 },
    bank_account: { type: 'partial', prefixLength: 0, suffixLength: 4 },
    name: { type: 'partial', prefixLength: 1, suffixLength: 0 },
  },
  encryptionRequired: true,
  accessLogRequired: true,
};

class PrivacyProtectionService {
  maskField(value: string, fieldName: string): string {
    const rule = this.config.maskRules[fieldName];
    
    if (!rule || rule.type === 'none' || !value) {
      return value;
    }
    
    if (rule.type === 'full') {
      return '*'.repeat(value.length);
    }
    
    const prefix = value.substring(0, rule.prefixLength || 0);
    const suffix = value.substring(value.length - (rule.suffixLength || 0));
    const maskLength = value.length - (rule.prefixLength || 0) - (rule.suffixLength || 0);
    const mask = (rule.maskChar || '*').repeat(Math.max(0, maskLength));
    
    return prefix + mask + suffix;
  }
  
  maskObject(obj: any, fields: string[]): any {
    const result = { ...obj };
    
    for (const field of fields) {
      if (result[field] !== undefined) {
        result[field] = this.maskField(result[field], field);
      }
    }
    
    return result;
  }
  
  async checkPiiAccess(userId: bigint, field: string): Promise<boolean> {
    const user = await this.userService.findById(userId);
    const rolePermissions = await this.getRoleFieldPermissions(user.roleId);
    
    return rolePermissions.includes(field) || this.isPiiFieldOwner(userId, field);
  }
  
  async logPiiAccess(
    userId: bigint,
    targetUserId: bigint,
    fields: string[],
    action: 'read' | 'export' | 'modify'
  ): Promise<void> {
    await this.auditService.log({
      eventType: 'PII_ACCESS',
      actorId: userId,
      targetId: targetUserId,
      fields,
      action,
      timestamp: new Date(),
    });
  }
}
```

```typescript
interface CrossBorderTransferConfig {
  enabled: boolean;
  requiresApproval: boolean;
  approvedDestinations: string[];
  restrictedDataTypes: string[];
  transferMechanisms: ('scc' | 'bc' | 'ad' | 'sa')[];
}

class CrossBorderTransferService {
  async validateTransfer(
    sourceRegion: string,
    targetRegion: string,
    dataTypes: string[]
  ): Promise<TransferValidationResult> {
    if (sourceRegion === 'cn' && targetRegion !== 'cn') {
      return {
        allowed: false,
        reason: '中国数据不得出境',
        requiresApproval: true,
        approvalLevel: 'regulatory',
      };
    }
    
    if (sourceRegion === 'eu' && !this.isValidEuDestination(targetRegion)) {
      const hasAdequacyDecision = await this.checkAdequacyDecision(targetRegion);
      const hasSCC = await this.checkStandardContractualClauses(sourceRegion, targetRegion);
      
      if (!hasAdequacyDecision && !hasSCC) {
        return {
          allowed: false,
          reason: '向该国家传输数据需要充分保护措施',
          requiresApproval: true,
          approvalLevel: 'dpo',
        };
      }
    }
    
    const sensitiveTypes = this.identifySensitiveDataTypes(dataTypes);
    if (sensitiveTypes.length > 0) {
      return {
        allowed: true,
        reason: '包含敏感数据，需要额外保护措施',
        requiresApproval: true,
        approvalLevel: 'security',
        additionalRequirements: [
          '加密传输',
          '访问日志记录',
          '数据处理协议签署',
        ],
      };
    }
    
    return {
      allowed: true,
      reason: '数据传输符合要求',
      requiresApproval: false,
    };
  }
  
  async executeTransfer(
    transfer: CrossBorderTransfer
  ): Promise<void> {
    const validation = await this.validateTransfer(
      transfer.sourceRegion,
      transfer.targetRegion,
      transfer.dataTypes
    );
    
    if (!validation.allowed) {
      throw new CrossBorderTransferDeniedError(validation.reason);
    }
    
    if (validation.requiresApproval) {
      await this.requestApproval(transfer, validation);
    }
    
    await this.performTransfer(transfer);
    
    await this.logTransfer(transfer);
  }
  
  private isValidEuDestination(region: string): boolean {
    const euDestinations = [
      'eu', 'us', 'ca', 'jp', 'uk', 'ch', 'kr', 
      'nz', 'au', 'sg'
    ];
    return euDestinations.includes(region);
  }
}
```
