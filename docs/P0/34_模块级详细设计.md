## 一、文档目的

本文档定义 MOY 系统各核心模块的详细设计，作为后端开发基线，用于：

1. 明确各模块的分层架构和职责边界
2. 定义服务层、领域层、仓储层的实现规范
3. 明确DTO/VO/Entity/Aggregate的边界
4. 定义状态机、权限、审计等横切关注点的实现位置
5. 定义模块间依赖关系和扩展点

## 二、分层架构规范

### 2.1 分层定义

```
┌─────────────────────────────────────────────────────────────┐
│                    Controller Layer                          │
│  职责：请求验证、响应格式化、权限校验入口                      │
└─────────────────────────────────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐
│                    Application Service Layer                  │
│  职责：用例编排、事务管理、权限校验、DTO转换                  │
└─────────────────────────────────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐
│                    Domain Service Layer                       │
│  职责：业务规则、状态机、领域事件                              │
└─────────────────────────────────────────────────────────────┘
                            │
```

### 2.2 各层职责矩阵

| Application Service | 用例编排、事务管理、权限校验、DTO转换 | 业务规则、状态机 |
| Domain Service | 业务规则、状态机、领域事件 | 数据库操作、外部调用 |

#### 3.2.2 Application Service层

```typescript
@Injectable()
export class CustomerService {
  constructor(
    private readonly customerDomain: CustomerDomainService,
    private readonly customerRepo: CustomerRepository,
    private readonly eventBus: EventBus,
  ) {}

  @Transactional()
  async create(orgId: bigint, userId: bigint, dto: CreateCustomerDto): Promise<Customer> {
    await this.customerDomain.checkDuplicate(orgId, dto.phone, dto.email);
    const customer = await this.customerDomain.createCustomer({
      orgId,
      name: dto.name,
      phone: dto.phone,
      email: dto.email,
      level: dto.level || 'C',
      source: dto.source || 'other',
      ownerId: dto.ownerId || userId,
      status: 'potential',
    });
    const saved = await this.customerRepo.save(customer);
    this.eventBus.publish(new CustomerCreatedEvent(saved.id, orgId, userId));
    return saved;
  }

  @Transactional()
  async update(orgId: bigint, id: bigint, userId: bigint, dto: UpdateCustomerDto): Promise<Customer> {
    const customer = await this.customerRepo.findById(orgId, id);
    if (!customer) {
      throw new NotFoundException('客户不存在');
    }
    const updated = await this.customerDomain.updateCustomer(customer, dto);
    const saved = await this.customerRepo.save(updated);
    this.eventBus.publish(new CustomerUpdatedEvent(saved.id, orgId, userId));
    return saved;
  }
}
```

#### 3.2.4 Repository层

```typescript
@Injectable()
export class CustomerRepository {
  constructor(private readonly dataSource: DataSource) {}

  async findById(orgId: bigint, id: bigint): Promise<Customer | null> {
    return this.dataSource.getRepository(Customer).findOne({
      where: { orgId, id },
      relations: ['owner', 'tags'],
    });
  }

  async findAll(orgId: bigint, query: CustomerListQuery): Promise<PaginatedResult<Customer>> {
    const qb = this.dataSource.getRepository(Customer).createQueryBuilder('customer');
    qb.where('customer.orgId = :orgId', { orgId });
    if (query.dataScope === 'self') {
      qb.andWhere('customer.ownerId = :userId', { userId: query.userId });
    } else if (query.dataScope === 'department') {
      qb.andWhere('customer.ownerId IN (:...userIds)', { userIds: query.departmentUserIds });
    }
    if (query.keyword) {
      qb.andWhere('customer.name LIKE :keyword', { keyword: `%${query.keyword}%` });
    }
    if (query.status) {
      qb.andWhere('customer.status = :status', { status: query.status });
    }
    qb.skip((query.page - 1) * query.pageSize).take(query.pageSize);
    const [items, total] = await qb.getManyAndCount();
    return { items, total, page: query.page, pageSize: query.pageSize };
  }

  async save(customer: Customer): Promise<Customer> {
    return this.dataSource.getRepository(Customer).save(customer);
  }
}
```

```typescript
@Entity('customers')
export class Customer {
  @PrimaryGeneratedColumn('increment', { type: 'bigint' })
  id: bigint;

  @Column({ type: 'bigint', name: 'org_id' })
  orgId: bigint;

  @Column({ length: 100 })
  name: string;

  @Column({ length: 20, nullable: true })
  phone: string;

  @Column({ length: 100, nullable: true })
  email: string;

  @Column({ length: 200, nullable: true })
  company: string;

  @Column({ type: 'enum', enum: CustomerLevel, default: CustomerLevel.C })
  level: CustomerLevel;

  @Column({ type: 'enum', enum: CustomerStatus, default: CustomerStatus.POTENTIAL })
  status: CustomerStatus;

  @Column({ type: 'enum', enum: CustomerSource, default: CustomerSource.OTHER })
  source: CustomerSource;

  @Column({ type: 'bigint', name: 'owner_id' })
  ownerId: bigint;

  @ManyToOne(() => User)
  @JoinColumn({ name: 'owner_id' })
  owner: User;

  @ManyToMany(() => Tag)
  @JoinTable({ name: 'customer_tags' })
  tags: Tag[];

  @CreateDateColumn({ name: 'created_at' })
  createdAt: Date;

  @UpdateDateColumn({ name: 'updated_at' })
  updatedAt: Date;

  update(data: Partial<Customer>): Customer {
    if (data.name) this.name = data.name;
    if (data.phone) this.phone = data.phone;
    if (data.email) this.email = data.email;
    if (data.level) this.level = data.level;
    return this;
  }

  static create(data: CreateCustomerData): Customer {
    const customer = new Customer();
    customer.orgId = data.orgId;
    customer.name = data.name;
    customer.phone = data.phone;
    customer.email = data.email;
    customer.level = data.level;
    customer.source = data.source;
    customer.ownerId = data.ownerId;
    customer.status = data.status;
    return customer;
  }
}
```

### 4.2 状态机实现

```typescript
@Injectable()
export class LeadStateMachine {
  private readonly transitions: Map<LeadStatus, LeadStatus[]> = new Map([
    [LeadStatus.NEW, [LeadStatus.ASSIGNED, LeadStatus.INVALID]],
    [LeadStatus.ASSIGNED, [LeadStatus.FOLLOWING, LeadStatus.INVALID]],
    [LeadStatus.FOLLOWING, [LeadStatus.CONVERTED, LeadStatus.INVALID]],
    [LeadStatus.CONVERTED, []],
    [LeadStatus.INVALID, []],
  ]);

  canTransition(from: LeadStatus, to: LeadStatus): boolean {
    const allowed = this.transitions.get(from) || [];
    return allowed.includes(to);
  }

  getAvailableTransitions(current: LeadStatus): LeadStatus[] {
    return this.transitions.get(current) || [];
  }

  isFinalState(status: LeadStatus): boolean {
    return status === LeadStatus.CONVERTED || status === LeadStatus.INVALID;
  }
}

@Injectable()
export class LeadDomainService {
  constructor(private readonly stateMachine: LeadStateMachine) {}

  async transitionStatus(lead: Lead, targetStatus: LeadStatus, context: TransitionContext): Promise<Lead> {
    if (!this.stateMachine.canTransition(lead.status, targetStatus)) {
      throw new BusinessException('060602', '线索状态流转不允许');
    }

    switch (targetStatus) {
      case LeadStatus.ASSIGNED:
        if (!context.assigneeId) {
          throw new BusinessException('060504', '分配线索需指定负责人');
        }
        lead.assignTo(context.assigneeId);
        break;

      case LeadStatus.CONVERTED:
        if (!context.customerId && !context.opportunityId) {
          throw new BusinessException('060503', '转化需关联客户或商机');
        }
        lead.convert(context.customerId, context.opportunityId);
        break;

      case LeadStatus.INVALID:
        if (!context.reason) {
          throw new BusinessException('060505', '标记无效需填写原因');
        }
        lead.markInvalid(context.reason);
        break;

      case LeadStatus.FOLLOWING:
        lead.startFollowing();
        break;
    }

    return lead;
  }
}
```

```typescript
export class OpportunityStateMachine {
  private readonly stageFlow: OpportunityStage[] = [
    OpportunityStage.INITIAL,
    OpportunityStage.REQUIREMENT,
    OpportunityStage.PROPOSAL,
    OpportunityStage.NEGOTIATION,
    OpportunityStage.WON,
    OpportunityStage.LOST,

    [OpportunityStage.INITIAL, [OpportunityStage.REQUIREMENT, OpportunityStage.LOST]],
    [OpportunityStage.REQUIREMENT, [OpportunityStage.PROPOSAL, OpportunityStage.INITIAL, OpportunityStage.LOST]],
    [OpportunityStage.PROPOSAL, [OpportunityStage.NEGOTIATION, OpportunityStage.REQUIREMENT, OpportunityStage.LOST]],
    [OpportunityStage.NEGOTIATION, [OpportunityStage.WON, OpportunityStage.PROPOSAL, OpportunityStage.LOST]],
    [OpportunityStage.WON, []],
    [OpportunityStage.LOST, []],
  ]);

  canTransition(from: OpportunityStage, to: OpportunityStage): boolean {
    const allowed = this.transitions.get(from) || [];
    return allowed.includes(to);
  }

  getProbability(stage: OpportunityStage): number {
    const probabilities: Record<OpportunityStage, number> = {
      [OpportunityStage.INITIAL]: 10,
      [OpportunityStage.REQUIREMENT]: 30,
      [OpportunityStage.PROPOSAL]: 50,
      [OpportunityStage.NEGOTIATION]: 70,
      [OpportunityStage.WON]: 100,
      [OpportunityStage.LOST]: 0,
    };
    return probabilities[stage];
```

### 6.2 WebSocket集成

```typescript
@WebSocketGateway({ namespace: '/conversation' })
export class ConversationGateway implements OnGatewayConnection, OnGatewayDisconnect {
  constructor(
    private readonly conversationService: ConversationService,
    private readonly authService: AuthService,
  ) {}

  async handleConnection(client: Socket) {
    const token = client.handshake.auth.token;
    const user = await this.authService.verifyToken(token);
    if (!user) {
      client.disconnect();
      return;
    }
    client.data.user = user;
    await this.conversationService.handleAgentOnline(user.id);
  }

  async handleDisconnect(client: Socket) {
    const user = client.data.user;
    if (user) {
      await this.conversationService.handleAgentOffline(user.id);
    }
  }

  @SubscribeMessage('send_message')
  async handleMessage(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: SendMessageDto,
  ) {
    const user = client.data.user;
    return this.conversationService.sendMessage(user.orgId, user.id, data);
  }
}
```

```typescript
export class TicketSlaService {
  constructor(private readonly configService: ConfigService) {}

  calculateDueDates(ticket: Ticket): { responseDue: Date; resolveDue: Date } {
    const slaConfig = this.configService.get('ticket.sla');
    const now = new Date();

    return {
      responseDue: this.addBusinessHours(now, slaConfig.responseHours),
      resolveDue: this.addBusinessHours(now, slaConfig.resolveHours),
    };
  }

  checkSlaBreached(ticket: Ticket): { responseBreached: boolean; resolveBreached: boolean } {
    const now = new Date();
    return {
      responseBreached: ticket.responseDue && now > ticket.responseDue,
      resolveBreached: ticket.resolveDue && now > ticket.resolveDue,
    };
  }
}
```

### 9.1 扩展点清单

### 9.2 扩展点实现示例

```typescript
interface AssignmentRule {
  name: string;
  execute(context: AssignmentContext): Promise<bigint>;
}

@Injectable()
export class RoundRobinAssignmentRule implements AssignmentRule {
  name = 'round_robin';
  
  async execute(context: AssignmentContext): Promise<bigint> {
    // 轮询分配逻辑
  }
}

@Injectable()
export class WorkloadAssignmentRule implements AssignmentRule {
  name = 'workload';
  
  async execute(context: AssignmentContext): Promise<bigint> {
    // 按工作量分配逻辑
  }
}

@Injectable()
export class AssignmentRuleFactory {
  private readonly rules: Map<string, AssignmentRule>;

  getRule(name: string): AssignmentRule {
    return this.rules.get(name);
  }
}
```
