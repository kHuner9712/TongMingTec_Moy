## 二、多租户隔离实现策略

### 2.1 隔离策略概述

| 层级 | 隔离方式 | 实现位置 |
|------|----------|----------|
| 数据库层 | 共享数据库，org_id字段隔离 | 所有表包含org_id字段 |
| 应用层 | 强制org_id过滤 | Repository层 + 守卫 |
| 缓存层 | org_id前缀隔离 | 缓存Key生成器 |
| 文件存储 | org_id目录隔离 | 文件上传服务 |

### 2.4 缓存层隔离

```typescript
@Injectable()
export class CacheService {
  private readonly prefix = 'moy';

  buildKey(orgId: bigint, key: string): string {
    return `${this.prefix}:org:${orgId}:${key}`;
  }

  async get<T>(orgId: bigint, key: string): Promise<T | null> {
    const fullKey = this.buildKey(orgId, key);
    const data = await this.redis.get(fullKey);
    return data ? JSON.parse(data) : null;
  }

  async set(orgId: bigint, key: string, value: unknown, ttl?: number): Promise<void> {
    const fullKey = this.buildKey(orgId, key);
    const data = JSON.stringify(value);
    if (ttl) {
      await this.redis.setex(fullKey, ttl, data);
    } else {
      await this.redis.set(fullKey, data);
    }
  }

  async deleteByPattern(orgId: bigint, pattern: string): Promise<void> {
    const fullPattern = this.buildKey(orgId, pattern);
    const keys = await this.redis.keys(fullPattern);
    if (keys.length > 0) {
      await this.redis.del(...keys);
    }
  }
}
```

## 三、数据权限注入策略

### 3.1 数据权限模型

| 范围类型 | 说明 | SQL条件 |
|----------|------|----------|
| all | 全部数据 | 无额外条件 |
| department | 本部门 | department_id = :deptId |
| department_and_sub | 本部门及下级 | department_id IN (:...deptIds) |
| self | 仅自己 | owner_id = :userId |
| custom | 自定义 | id IN (:...ids) |

### 3.2 权限注入实现

```typescript
@Injectable()
export class DataScopeService {
  constructor(
    private readonly userRepo: UserRepository,
    private readonly deptRepo: DepartmentRepository,
  ) {}

  async getDataScope(userId: bigint, resource: string): Promise<DataScope> {
    const user = await this.userRepo.findById(userId);
    const role = user.roles.find(r => r.permissions.some(p => p.startsWith(resource)));

    if (!role) {
      return { type: 'self', userId };
    }

    const scopeConfig = role.dataScopeConfig?.[resource];
    
    switch (scopeConfig?.type) {
      case 'all':
        return { type: 'all' };
      case 'department':
        return { type: 'department', departmentId: user.departmentId };
      case 'department_and_sub':
        const subDeptIds = await this.deptRepo.getSubDepartmentIds(user.departmentId);
        return { type: 'department_and_sub', departmentIds: [user.departmentId, ...subDeptIds] };
      case 'self':
        return { type: 'self', userId };
      case 'custom':
        return { type: 'custom', entityIds: scopeConfig.entityIds };
      default:
        return { type: 'self', userId };
    }
  }
}
```

### 3.3 Repository层应用

```typescript
@Injectable()
export class CustomerRepository {
  constructor(
    private readonly dataSource: DataSource,
    private readonly dataScopeService: DataScopeService,
  ) {}

  async findAllWithScope(
    orgId: bigint,
    userId: bigint,
    query: CustomerListQuery,
  ): Promise<PaginatedResult<Customer>> {
    const dataScope = await this.dataScopeService.getDataScope(userId, 'customer');
    const qb = this.dataSource.getRepository(Customer).createQueryBuilder('customer');

    // 多租户隔离
    qb.where('customer.orgId = :orgId', { orgId });

    // 数据权限注入
    this.applyDataScope(qb, dataScope);

    // 其他筛选条件
    if (query.keyword) {
      qb.andWhere('customer.name LIKE :keyword', { keyword: `%${query.keyword}%` });
    }

    return this.paginate(qb, query);
  }

  private applyDataScope(qb: SelectQueryBuilder<Customer>, scope: DataScope): void {
    switch (scope.type) {
      case 'all':
        break;
      case 'department':
        qb.andWhere('customer.departmentId = :deptId', { deptId: scope.departmentId });
        break;
      case 'department_and_sub':
        qb.andWhere('customer.departmentId IN (:...deptIds)', { deptIds: scope.departmentIds });
        break;
      case 'self':
        qb.andWhere('customer.ownerId = :userId', { userId: scope.userId });
        break;
      case 'custom':
        qb.andWhere('customer.id IN (:...ids)', { ids: scope.entityIds });
        break;
    }
  }
}
```

```typescript
@Injectable()
export class AuditInterceptor implements NestInterceptor {
  constructor(private readonly auditService: AuditService) {}

  intercept(context: ExecutionContext, next: CallHandler): Observable<unknown> {
    const request = context.switchToHttp().getRequest();
    const { method, url, body, params } = request;
    const user = request.user;

    if (!this.shouldAudit(method, url)) {
      return next.handle();
    }

    return next.handle().pipe(
      tap((response) => {
        this.auditService.log({
          orgId: user.orgId,
          entityType: this.extractEntityType(url),
          entityId: this.extractEntityId(params, response),
          action: this.extractAction(method),
          newValue: response,
          operatorId: user.id,
          operatorName: user.name,
          ipAddress: request.ip,
          userAgent: request.headers['user-agent'],
        });
      }),
    );
  }

  private shouldAudit(method: string, url: string): boolean {
    const auditMethods = ['POST', 'PUT', 'PATCH', 'DELETE'];
    const excludePatterns = ['/auth/', '/health', '/metrics'];
    
    return auditMethods.includes(method) && 
           !excludePatterns.some(p => url.includes(p));
  }
}
```

### 4.4 审计装饰器

```typescript
export function AuditLog(entity: string, action: string) {
  return (target: unknown, propertyKey: string, descriptor: PropertyDescriptor) => {
    const originalMethod = descriptor.value;

    descriptor.value = async function (...args: unknown[]) {
      const request = args.find(a => a?.constructor?.name === 'IncomingMessage');
      const user = request?.user;
      const orgId = request?.orgId;
      const entityId = args.find(a => typeof a === 'object' && a?.id)?.id;

      // 获取变更前数据
      let oldValue: unknown = null;
      if (action !== 'create' && entityId) {
        oldValue = await this.getOldValue?.(orgId, entityId);
      }

      // 执行原方法
      const result = await originalMethod.apply(this, args);

      // 记录审计日志
      await this.auditService.log({
        orgId,
        entityType: entity,
        entityId: entityId || result?.id,
        action,
        oldValue,
        newValue: result,
        operatorId: user?.id,
        operatorName: user?.name,
        ipAddress: request?.ip,
        userAgent: request?.headers?.['user-agent'],
      });

      return result;
    };
  };
}

  return this.customerService.create(orgId, user.id, dto);
}
```

## 五、敏感字段脱敏策略

### 5.1 脱敏规则定义

| 字段类型 | 脱敏规则 | 示例 |
|----------|----------|------|
| 手机号 | 中间4位 | 138****1234 |
| 邮箱 | @前保留首字符 | a***@example.com |
| 身份证 | 中间10位 | 110***********1234 |
| 银行卡 | 中间12位 | 6222************1234 |
| 姓名 | 保留首字符 | 张** |

### 5.2 脱敏服务实现

```typescript
@Injectable()
export class MaskService {
  maskPhone(phone: string): string {
    if (!phone || phone.length < 7) return phone;
    return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2');
  }

  maskEmail(email: string): string {
    if (!email || !email.includes('@')) return email;
    const [local, domain] = email.split('@');
    const maskedLocal = local.charAt(0) + '***';
    return `${maskedLocal}@${domain}`;
  }

  maskIdCard(idCard: string): string {
    if (!idCard || idCard.length < 15) return idCard;
    return idCard.replace(/(\d{3})\d{10}(\d{4})/, '$1***********$2');
  }

  maskBankCard(cardNo: string): string {
    if (!cardNo || cardNo.length < 16) return cardNo;
    return cardNo.replace(/(\d{4})\d{12}(\d{4})/, '$1************$2');
  }

  maskName(name: string): string {
    if (!name) return name;
    return name.charAt(0) + '*'.repeat(Math.min(name.length - 1, 2));
  }
}
```

### 5.3 响应拦截器

```typescript
@Injectable()
export class MaskInterceptor implements NestInterceptor {
  constructor(private readonly maskService: MaskService) {}

  intercept(context: ExecutionContext, next: CallHandler): Observable<unknown> {
    return next.handle().pipe(
      map((data) => this.maskSensitiveFields(data)),
    );
  }

  private maskSensitiveFields(data: unknown): unknown {
    if (!data || typeof data !== 'object') return data;

    if (Array.isArray(data)) {
      return data.map(item => this.maskSensitiveFields(item));
    }

    const masked = { ...data };
    const sensitiveFields = {
      phone: this.maskService.maskPhone,
      email: this.maskService.maskEmail,
      idCard: this.maskService.maskIdCard,
      bankCard: this.maskService.maskBankCard,
    };

    for (const [field, maskFn] of Object.entries(sensitiveFields)) {
      if (masked[field]) {
        masked[field] = maskFn(masked[field]);
      }
    }

    return masked;
  }
}
```

```typescript
@Injectable()
export class IdempotentService {
  constructor(private readonly redis: RedisService) {}

  async acquire(orgId: bigint, key: string, ttl: number = 86400): Promise<boolean> {
    const fullKey = `idempotent:${orgId}:${key}`;
    const result = await this.redis.setnx(fullKey, 'pending');
    if (result) {
      await this.redis.expire(fullKey, ttl);
    }
    return result === 1;
  }

  async complete(orgId: bigint, key: string, result: unknown): Promise<void> {
    const fullKey = `idempotent:${orgId}:${key}`;
    await this.redis.setex(fullKey, 86400, JSON.stringify({ status: 'completed', result }));
  }

  async getResult(orgId: bigint, key: string): Promise<unknown | null> {
    const fullKey = `idempotent:${orgId}:${key}`;
    const data = await this.redis.get(fullKey);
    if (data) {
      const parsed = JSON.parse(data);
      if (parsed.status === 'completed') {
        return parsed.result;
      }
    }
    return null;
  }
}

// 幂等装饰器
export function Idempotent(keyExtractor: (...args: unknown[]) => string) {
  return (target: unknown, propertyKey: string, descriptor: PropertyDescriptor) => {
    const originalMethod = descriptor.value;

    descriptor.value = async function (...args: unknown[]) {
      const idempotentService = this.idempotentService;
      const orgId = args.find(a => typeof a === 'bigint' && a > 0);
      const key = keyExtractor(...args);

      // 检查是否已有结果
      const existingResult = await idempotentService.getResult(orgId, key);
      if (existingResult) {
        return existingResult;
      }

      // 获取幂等锁
      const acquired = await idempotentService.acquire(orgId, key);
      if (!acquired) {
        // 等待其他请求完成
        await new Promise(resolve => setTimeout(resolve, 100));
        return idempotentService.getResult(orgId, key);
      }

      try {
        const result = await originalMethod.apply(this, args);
        await idempotentService.complete(orgId, key, result);
        return result;
      } catch (error) {
        await idempotentService.release(orgId, key);
        throw error;
      }
    };
  };
}

// 使用示例
@Post()
@Idempotent((dto, orgId) => `customer:create:${dto.phone || dto.email}`)
async create(@Body() dto: CreateCustomerDto, @OrgId() orgId: bigint) {
  return this.customerService.create(orgId, dto);
}
```

```typescript
@Injectable()
export class ExportService {
  constructor(
    private readonly taskRepo: ExportTaskRepository,
    private readonly storageService: FileStorageService,
    private readonly queueService: QueueService,
  ) {}

  async createExportTask(
    orgId: bigint,
    userId: bigint,
    type: ExportType,
    params: ExportParams,
  ): Promise<ExportTask> {
    const task = await this.taskRepo.create({
      orgId,
      userId,
      type,
      status: 'pending',
      params,
    });

    await this.queueService.add('export', {
      taskId: task.id,
      orgId,
      type,
      params,
    });

    return task;
  }

  async processExportTask(taskId: bigint): Promise<void> {
    const task = await this.taskRepo.findById(taskId);
    
    try {
      await this.taskRepo.updateStatus(taskId, 'processing');

      const data = await this.fetchData(task.orgId, task.type, task.params);
      const filePath = await this.generateFile(task.orgId, task.type, data);

      await this.taskRepo.update(taskId, {
        status: 'completed',
        filePath,
        fileSize: await this.storageService.getSize(filePath),
        totalCount: data.length,
        completedAt: new Date(),
      });
    } catch (error) {
      await this.taskRepo.update(taskId, {
        status: 'failed',
        errorMessage: error.message,
      });
    }
  }
}
```

### 8.2 文件上传服务

```typescript
@Injectable()
export class FileUploadService {
  constructor(
    private readonly storageService: StorageService,
    private readonly virusScanService: VirusScanService,
  ) {}

  @Transactional()
  async upload(
    orgId: bigint,
    userId: bigint,
    file: Express.Multer.File,
    options: UploadOptions = {},
  ): Promise<FileInfo> {
    // 1. 文件校验
    this.validateFile(file, options);

    // 2. 病毒扫描（如果启用）
    if (options.scanVirus) {
      await this.virusScanService.scan(file.buffer);
    }

    // 3. 生成存储路径
    const path = this.generatePath(orgId, file.originalname, options);

    // 4. 上传到存储
    const url = await this.storageService.upload(path, file.buffer, {
      contentType: file.mimetype,
      isPrivate: options.isPrivate,
    });

    // 5. 记录文件信息
    const fileInfo = await this.fileRepo.create({
      orgId,
      userId,
      name: file.originalname,
      path,
      size: file.size,
      mimeType: file.mimetype,
      url,
      isPrivate: options.isPrivate || false,
    });

    return fileInfo;
  }

  private validateFile(file: Express.Multer.File, options: UploadOptions): void {
    const maxSize = options.maxSize || 10 * 1024 * 1024; // 默认10MB
    if (file.size > maxSize) {
      throw new BusinessException('FILE_001', `文件大小超过限制（${maxSize / 1024 / 1024}MB）`);
    }

    if (options.allowedTypes) {
      const ext = path.extname(file.originalname).toLowerCase();
      if (!options.allowedTypes.includes(ext)) {
        throw new BusinessException('FILE_002', `不支持的文件类型：${ext}`);
      }
    }
  }

  private generatePath(orgId: bigint, filename: string, options: UploadOptions): string {
    const date = new Date();
    const ext = path.extname(filename);
    const uuid = uuidv4();
    const prefix = options.prefix || 'files';
    
    return `orgs/${orgId}/${prefix}/${date.getFullYear()}/${date.getMonth() + 1}/${uuid}${ext}`;
  }
}
```

```typescript
@Injectable()
export class WebhookService {
  constructor(
    private readonly webhookRepo: WebhookRepository,
    private readonly httpService: HttpService,
  ) {}

  async trigger(orgId: bigint, event: string, payload: unknown): Promise<void> {
    const webhooks = await this.webhookRepo.findByEvent(orgId, event);

    for (const webhook of webhooks) {
      await this.sendWebhook(webhook, event, payload);
    }
  }

  private async sendWebhook(
    webhook: Webhook,
    event: string,
    payload: unknown,
  ): Promise<void> {
    const body = {
      event,
      timestamp: Date.now(),
      data: payload,
    };

    const signature = this.generateSignature(webhook.secret, body);

    try {
      const response = await this.httpService.post(webhook.url, body, {
        headers: {
          'Content-Type': 'application/json',
          'X-Webhook-Signature': signature,
        },
        timeout: 10000,
      }).toPromise();

      await this.logWebhook(webhook, event, body, response.status, response.data);
    } catch (error) {
      await this.logWebhook(webhook, event, body, null, error.message);
      // 重试逻辑
      await this.retryWebhook(webhook, event, body);
    }
  }

  private generateSignature(secret: string, body: unknown): string {
    const hmac = crypto.createHmac('sha256', secret);
    hmac.update(JSON.stringify(body));
    return `sha256=${hmac.digest('hex')}`;
  }
}
```

```typescript
@Injectable()
export class QueueService {
  constructor(private readonly redis: RedisService) {}

  async add(queue: string, data: unknown, options?: QueueOptions): Promise<void> {
    const message = JSON.stringify({
      id: uuidv4(),
      data,
      createdAt: Date.now(),
      ...options,
    });

    await this.redis.xadd(queue, '*', { message });
  }

  async consume(queue: string, handler: (data: unknown) => Promise<void>): Promise<void> {
    while (true) {
      const messages = await this.redis.xread('BLOCK', 0, 'STREAMS', queue, '$');
      
      for (const [stream, entries] of messages) {
        for (const [id, fields] of entries) {
          try {
            const message = JSON.parse(fields.message);
            await handler(message.data);
            await this.redis.xack(queue, 'moy-group', id);
          } catch (error) {
            console.error(`Failed to process message ${id}:`, error);
          }
        }
      }
    }
  }
}
```

```typescript
@Injectable()
export class AIGateway {
  private readonly providers: Map<string, AIProvider> = new Map();
  private readonly rateLimiter: RateLimiter;

  constructor(
    private readonly configService: ConfigService,
    private readonly circuitBreaker: CircuitBreaker,
  ) {
    this.initProviders();
  }

  private initProviders(): void {
    const config = this.configService.get('ai');
    
    this.providers.set('deepseek', new DeepSeekProvider(config.deepseek));
    this.providers.set('qwen', new QwenProvider(config.qwen));
  }

  async chat(request: ChatRequest): Promise<ChatResponse> {
    const provider = this.selectProvider();
    
    // 限流检查
    await this.rateLimiter.acquire(request.orgId);

    // 熔断检查
    if (this.circuitBreaker.isOpen(provider.name)) {
      throw new AIServiceUnavailableError('AI服务暂时不可用');
    }

    try {
      const response = await this.circuitBreaker.execute(
        provider.name,
        () => provider.chat(request),
      );
      
      return response;
    } catch (error) {
      // 降级处理
      return this.fallback(request, error);
    }
  }

  private selectProvider(): AIProvider {
    const primary = this.configService.get('ai.primaryProvider');
    const provider = this.providers.get(primary);
    
    if (!provider || this.circuitBreaker.isOpen(primary)) {
      const fallback = this.configService.get('ai.fallbackProvider');
      return this.providers.get(fallback);
    }
    
    return provider;
  }

  private async fallback(request: ChatRequest, error: Error): Promise<ChatResponse> {
    // 记录降级事件
    await this.auditService.logAIFallback(request, error);
    
    // 返回降级响应
    return {
      content: '抱歉，AI服务暂时不可用，请稍后重试或联系人工客服。',
      isFallback: true,
    };
  }
}
```

```typescript
export class AIRateLimiter {
  constructor(private readonly redis: RedisService) {}

  async acquire(orgId: bigint): Promise<void> {
    const key = `ai:rate:${orgId}:${dayjs().format('YYYYMMDD')}`;
    const current = await this.redis.incr(key);
    
    if (current === 1) {
      await this.redis.expire(key, 86400);
    }

    const limit = await this.getQuota(orgId);
    
    if (current > limit) {
      throw new AIQuotaExceededError(`AI配额已用尽（${limit}次/天）`);
    }
  }

  private async getQuota(orgId: bigint): Promise<number> {
    const org = await this.orgRepo.findById(orgId);
    return org.aiQuotaDaily || 1000;
  }
}
```

## 十三、降级与熔断策略

### 13.1 熔断器实现

```typescript
@Injectable()
export class CircuitBreaker {
  private readonly states: Map<string, CircuitState> = new Map();
  
  private readonly failureThreshold = 5;
  private readonly successThreshold = 3;
  private readonly timeout = 30000; // 30秒

  isOpen(service: string): boolean {
    const state = this.states.get(service);
    if (!state) return false;

    if (state.status === 'open') {
      if (Date.now() - state.lastFailure > this.timeout) {
        this.states.set(service, { ...state, status: 'half-open' });
        return false;
      }
      return true;
    }

    return false;
  }

  async execute<T>(service: string, fn: () => Promise<T>): Promise<T> {
    try {
      const result = await fn();
      this.recordSuccess(service);
      return result;
    } catch (error) {
      this.recordFailure(service);
      throw error;
    }
  }

  private recordSuccess(service: string): void {
    const state = this.states.get(service);
    if (!state) return;

    if (state.status === 'half-open') {
      state.successCount++;
      if (state.successCount >= this.successThreshold) {
        this.states.delete(service);
      }
    }
  }

  private recordFailure(service: string): void {
    let state = this.states.get(service);
    if (!state) {
      state = { status: 'closed', failureCount: 0, successCount: 0, lastFailure: 0 };
    }

    state.failureCount++;
    state.lastFailure = Date.now();

    if (state.failureCount >= this.failureThreshold) {
      state.status = 'open';
    }

    this.states.set(service, state);
  }
}
```

```typescript
@Catch(BusinessException)
export class BusinessExceptionFilter implements ExceptionFilter {
  catch(exception: BusinessException, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse();

    response.status(this.getHttpStatus(exception.code)).json({
      code: exception.code,
      message: exception.message,
      details: exception.details,
      timestamp: new Date().toISOString(),
    });
  }

  private getHttpStatus(code: string): number {
    const prefix = code.substring(0, 2);
    const typeCode = code.substring(2, 4);

    if (prefix === '02' && typeCode === '02') return 401;
    if (prefix === '02' && typeCode === '03') return 403;
    if (typeCode === '04') return 404;
    if (typeCode === '05' || typeCode === '06') return 422;
    if (typeCode === '07' || typeCode === '08') return 409;

    return 400;
  }
}
```
