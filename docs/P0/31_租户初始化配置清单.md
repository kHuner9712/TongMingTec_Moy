## 一、文档目的

本文档定义 MOY 系统新租户（组织）初始化的完整配置清单，确保：

1. 租户初始化流程标准化
2. 必要数据自动创建
3. 默认配置合理可用
4. 初始化过程可追溯
5. 支持批量初始化

## 九、初始化脚本

### 9.1 TypeScript实现

```typescript
import { Injectable } from '@nestjs/common';
import { DataSource } from 'typeorm';

@Injectable()
export class TenantInitService {
  constructor(private dataSource: DataSource) {}

  async initTenant(dto: InitTenantDto): Promise<InitTenantResult> {
    const queryRunner = this.dataSource.createQueryRunner();
    await queryRunner.connect();
    await queryRunner.startTransaction();

    try {
      const orgId = await this.createOrganization(queryRunner, dto);
      await this.createRoles(queryRunner, orgId);
      const deptIds = await this.createDepartments(queryRunner, orgId);
      const userId = await this.createAdminUser(queryRunner, orgId, dto, deptIds.root);
      await this.createConfigs(queryRunner, orgId);

      if (dto.includeSampleData) {
        await this.createSampleData(queryRunner, orgId, userId, deptIds);
      }

      await queryRunner.commitTransaction();

      return {
        success: true,
        orgId,
        adminUserId: userId,
        message: '租户初始化成功',
      };
    } catch (error) {
      await queryRunner.rollbackTransaction();
      throw error;
    } finally {
      await queryRunner.release();
    }
  }

  private async createOrganization(queryRunner: QueryRunner, dto: InitTenantDto): Promise<bigint> {
    const result = await queryRunner.query(
      `INSERT INTO organizations (name, slug, status, plan, max_users, max_storage_mb, created_at)
       VALUES ($1, $2, 'active', 'trial', 10, 1024, NOW())
       RETURNING id`,
      [dto.orgName, this.generateSlug(dto.orgName)]
    );
    return result[0].id;
  }

  private async createRoles(queryRunner: QueryRunner, orgId: bigint): Promise<void> {
    const roles = this.getDefaultRoles();
    for (const role of roles) {
      await queryRunner.query(
        `INSERT INTO roles (org_id, code, name, description, is_system, permissions, created_at)
         VALUES ($1, $2, $3, $4, $5, $6, NOW())`,
        [orgId, role.code, role.name, role.description, role.isSystem, JSON.stringify(role.permissions)]
      );
    }
  }

  private async createDepartments(queryRunner: QueryRunner, orgId: bigint): Promise<DepartmentIds> {
    const rootResult = await queryRunner.query(
      `INSERT INTO departments (org_id, name, parent_id, status, created_at)
       VALUES ($1, '总经办', NULL, 'active', NOW())
       RETURNING id`,
      [orgId]
    );
    const rootId = rootResult[0].id;

    await queryRunner.query(
      `INSERT INTO departments (org_id, name, parent_id, status, created_at)
       VALUES ($1, '销售部', $2, 'active', NOW()),
              ($1, '客服部', $2, 'active', NOW())`,
      [orgId, rootId]
    );

    return { root: rootId };
  }

  private async createAdminUser(
    queryRunner: QueryRunner,
    orgId: bigint,
    dto: InitTenantDto,
    deptId: bigint
  ): Promise<bigint> {
    const hashedPassword = await bcrypt.hash(dto.adminPassword, 12);

    const userResult = await queryRunner.query(
      `INSERT INTO users (org_id, email, name, password_hash, status, department_id, created_at)
       VALUES ($1, $2, $3, $4, 'active', $5, NOW())
       RETURNING id`,
      [orgId, dto.adminEmail, dto.adminName, hashedPassword, deptId]
    );
    const userId = userResult[0].id;

    await queryRunner.query(
      `INSERT INTO user_roles (org_id, user_id, role_id, created_at)
       SELECT $1, $2, id, NOW()
       FROM roles WHERE org_id = $1 AND code = 'admin'`,
      [orgId, userId]
    );

    return userId;
  }

  private async createConfigs(queryRunner: QueryRunner, orgId: bigint): Promise<void> {
    const configs = this.getDefaultConfigs();
    for (const config of configs) {
      await queryRunner.query(
        `INSERT INTO system_configs (org_id, config_key, config_value, created_at)
         VALUES ($1, $2, $3, NOW())`,
        [orgId, config.key, JSON.stringify(config.value)]
      );
    }
  }

  private generateSlug(name: string): string {
    return name
      .toLowerCase()
      .replace(/[^a-z0-9\u4e00-\u9fa5]/g, '-')
      .substring(0, 50);
  }

  private getDefaultRoles(): RoleConfig[] {
    return [
      { code: 'admin', name: '系统管理员', description: '拥有所有权限', isSystem: true, permissions: ['*'] },
      { code: 'sales_manager', name: '销售经理', description: '管理销售团队', isSystem: false, permissions: ['customer:read', 'customer:write', 'lead:*', 'opportunity:*'] },
      { code: 'sales_rep', name: '销售代表', description: '销售人员', isSystem: false, permissions: ['customer:read', 'customer:write', 'lead:read', 'lead:write'] },
      { code: 'service_manager', name: '客服经理', description: '管理客服团队', isSystem: false, permissions: ['customer:read', 'conversation:*', 'ticket:*'] },
      { code: 'service_agent', name: '客服专员', description: '客服人员', isSystem: false, permissions: ['customer:read', 'conversation:read', 'conversation:write', 'ticket:read', 'ticket:write'] },
    ];
  }

  private getDefaultConfigs(): ConfigItem[] {
    return [
      { key: 'system.timezone', value: 'Asia/Shanghai' },
      { key: 'system.date_format', value: 'YYYY-MM-DD' },
      { key: 'system.time_format', value: 'HH:mm:ss' },
      { key: 'system.currency', value: 'CNY' },
      { key: 'system.language', value: 'zh-CN' },
      { key: 'lead.auto_assign', value: false },
      { key: 'ticket.sla_enabled', value: true },
      { key: 'ai.enabled', value: true },
    ];
  }
}
```
