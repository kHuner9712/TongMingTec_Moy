## 一、文档目的

本文档定义 MOY 系统的前端 UI 组件库规范，作为企业级 AI 原生客户管理系统的前端开发基线，用于：

1. 统一前端组件库选型和技术栈
2. 定义组件命名和使用规范
3. 规范主题、样式、交互一致性
4. 提高前端开发效率和代码质量
5. 确保用户体验一致性

```tsx
// components/business/StandardTable/index.tsx
import { Table, TableProps } from 'antd';
import type { ColumnsType } from 'antd/es/table';

interface StandardTableProps<T> extends Omit<TableProps<T>, 'pagination'> {
  columns: ColumnsType<T>;
  dataSource: T[];
  loading?: boolean;
  pagination?: {
    current: number;
    pageSize: number;
    total: number;
    onChange: (page: number, pageSize: number) => void;
  };
  rowKey: string | ((record: T) => string);
}

export function StandardTable<T extends Record<string, unknown>>({
  columns,
  dataSource,
  loading,
  pagination,
  rowKey,
  ...restProps
}: StandardTableProps<T>) {
  return (
    <Table<T>
      columns={columns}
      dataSource={dataSource}
      loading={loading}
      rowKey={rowKey}
      pagination={
        pagination
          ? {
              current: pagination.current,
              pageSize: pagination.pageSize,
              total: pagination.total,
              onChange: pagination.onChange,
              showSizeChanger: true,
              showQuickJumper: true,
              showTotal: (total) => `共 ${total} 条`,
              pageSizeOptions: ['10', '20', '50', '100'],
            }
          : false
      }
      scroll={{ x: 'max-content' }}
      {...restProps}
    />
  );
}
```

### 4.2 表单组件

#### 4.2.1 标准表单配置

```tsx
// components/business/StandardForm/index.tsx
import { Form, FormProps, Button, Space } from 'antd';
import { ReactNode } from 'react';

interface StandardFormProps extends FormProps {
  children: ReactNode;
  onSubmit: (values: unknown) => Promise<void>;
  submitText?: string;
  cancelText?: string;
  onCancel?: () => void;
  loading?: boolean;
  footer?: ReactNode;
}

export function StandardForm({
  children,
  onSubmit,
  submitText = '提交',
  cancelText = '取消',
  onCancel,
  loading,
  footer,
  ...restProps
}: StandardFormProps) {
  const [form] = Form.useForm();

  const handleSubmit = async (values: unknown) => {
    await onSubmit(values);
  };

  return (
    <Form form={form} layout="vertical" onFinish={handleSubmit} {...restProps}>
      {children}
      {footer !== null && (
        <Form.Item>
          <Space>
            <Button type="primary" htmlType="submit" loading={loading}>
              {submitText}
            </Button>
            {onCancel && (
              <Button onClick={onCancel}>{cancelText}</Button>
            )}
          </Space>
        </Form.Item>
      )}
    </Form>
  );
}
```

### 4.4 搜索筛选组件

```tsx
// components/business/SearchFilter/index.tsx
import { Form, Input, Select, Button, Space, Row, Col } from 'antd';
import { SearchOutlined, ReloadOutlined } from '@ant-design/icons';

interface FilterField {
  name: string;
  label: string;
  type: 'input' | 'select' | 'date' | 'dateRange';
  options?: { label: string; value: string | number }[];
  placeholder?: string;
}

interface SearchFilterProps {
  fields: FilterField[];
  onSearch: (values: Record<string, unknown>) => void;
  onReset: () => void;
  loading?: boolean;
}

export function SearchFilter({ fields, onSearch, onReset, loading }: SearchFilterProps) {
  const [form] = Form.useForm();

  const handleSearch = () => {
    const values = form.getFieldsValue();
    onSearch(values);
  };

  const handleReset = () => {
    form.resetFields();
    onReset();
  };

  return (
    <Form form={form} layout="inline">
      <Row gutter={16}>
        {fields.map((field) => (
          <Col key={field.name} span={6}>
            <Form.Item name={field.name} label={field.label}>
              {field.type === 'input' && (
                <Input placeholder={field.placeholder || `请输入${field.label}`} allowClear />
              )}
              {field.type === 'select' && (
                <Select
                  placeholder={field.placeholder || `请选择${field.label}`}
                  options={field.options}
                  allowClear
                />
              )}
              {field.type === 'date' && (
                <DatePicker style={{ width: '100%' }} placeholder={field.placeholder} />
              )}
              {field.type === 'dateRange' && (
                <RangePicker style={{ width: '100%' }} />
              )}
            </Form.Item>
          </Col>
        ))}
        <Col span={6}>
          <Space>
            <Button type="primary" icon={<SearchOutlined />} onClick={handleSearch} loading={loading}>
              搜索
            </Button>
            <Button icon={<ReloadOutlined />} onClick={handleReset}>
              重置
            </Button>
          </Space>
        </Col>
      </Row>
    </Form>
  );
}
```

### 8.2 缓动函数

```css
:root {
  --ease-in: cubic-bezier(0.4, 0, 1, 1);
  --ease-out: cubic-bezier(0, 0, 0.2, 1);
  --ease-in-out: cubic-bezier(0.4, 0, 0.2, 1);
}

/* 使用示例 */
.fade-enter {
  opacity: 0;
}
.fade-enter-active {
  opacity: 1;
  transition: opacity 300ms var(--ease-out);
}
.fade-exit {
  opacity: 1;
}
.fade-exit-active {
  opacity: 0;
  transition: opacity 300ms var(--ease-in);
}
```
